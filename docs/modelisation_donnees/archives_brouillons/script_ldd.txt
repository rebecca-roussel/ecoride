Sub Create_Tables()

DoCmd.RunSQL "CREATE TABLE UTILISATEUR(" & _
   "Id_UTILISATEUR COUNTER," & _
   "pseudo VARCHAR(50) NOT NULL," & _

   "role_passager LOGICAL NOT NULL," & _

   "statut VARCHAR(20) NOT NULL," & _
   "PRIMARY KEY(Id_UTILISATEUR)," & _

   "UNIQUE(email)" & _
");"

");"

   "incident_resolu LOGICAL NOT NULL," & _
   "Id_UTILISATEUR INT NOT NULL," & _

   "PRIMARY KEY(Id_COVOITURAGE)," & _

");"

   "PRIMARY KEY(Id_PARTICIPATION)," & _

   "Id_UTILISATEUR INT," & _
   "PRIMARY KEY(Id_UTILISATEUR)," & _

");"

   "Id_UTILISATEUR INT," & _

");"

   "Id_AVIS COUNTER," & _
   "note INT NOT NULL," & _
   "commentaire VARCHAR(1000)," & _
   "date_depot DATETIME NOT NULL," & _

");"
